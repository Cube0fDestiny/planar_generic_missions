feyrealm_goblins_clear_cut_modifier = {
	potential = {
		always = yes
	}

	trigger = {
		has_province_modifier = feyrealm_goblins_clear_cut_modifier_temp
	}
	
	hidden = yes
	
	#Province scope
	on_activation = {
	
	}
	
	on_deactivation = {
		trigger_switch = {
			on_trigger = has_terrain
			ancient_forest = {
				add_permanent_province_modifier = {
					name = cleared_ancient_forest_terrain
					duration = -1
					hidden = yes
				}
				add_permanent_province_modifier = {
					name = clear_cut_wasteland
					duration = -1
					hidden = yes
				}
			}
			forest = {
				add_permanent_province_modifier = {
					name = cleared_forest_terrain
					duration = -1
					hidden = yes
				}
				add_permanent_province_modifier = {
					name = clear_cut_wasteland
					duration = -1
					hidden = yes
				}
			}
			woods = {
				add_permanent_province_modifier = {
					name = cleared_woods_terrain
					duration = -1
					hidden = yes
				}
				add_permanent_province_modifier = {
					name = clear_cut_wasteland
					duration = -1
					hidden = yes
				}
			}
		}
		remove_province_triggered_modifier = feyrealm_goblins_clear_cut_modifier
		set_province_flag = clear_cut_wasteland_terrain
	}
}
