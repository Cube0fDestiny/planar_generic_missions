province_event = {
	id = planar_generic_missions_events.108
	title = planar_generic_missions_events.108.t
	picture = PRAYING_eventPicture
	desc = planar_generic_missions_events.108.d

	hidden = yes

	mean_time_to_happen = {
		months = 1
	}

	trigger = {
		owner = { has_country_flag = adapted_to_the_sahdow_plane }
		OR = {
			AND = {
				has_terrain = shadow_city_terrain
				NOT = { has_province_modifier = shadow_plane_adaptability_city }
			}
			AND = {
				has_terrain = shadow_swamp_terrain
				NOT = { has_province_modifier = shadow_plane_adaptability_swamp }
			}
			AND = {
				has_terrain = shadow_wall_terrain
				NOT = { has_province_modifier = shadow_plane_adaptability_battlefields }
			}
			AND = {
				has_terrain = fluxwastes_terrain
				NOT = { has_province_modifier = shadow_plane_adaptability_flux_wastes }
			}
		}
	}

	option = {
		name = planar_generic_missions_events.108.a
		trigger_switch = {
			on_trigger = has_terrain
			shadow_city_terrain = {
				add_province_modifier = {
					name = shadow_plane_adaptability_city
					duration = -1
				}
			}
			shadow_swamp_terrain = {
				add_province_modifier = {
					name = shadow_plane_adaptability_swamp
					duration = -1
				}
			}
			shadow_wall_terrain = {
				add_province_modifier = {
					name = shadow_plane_adaptability_battlefields
					duration = -1
				}
			}
			fluxwastes_terrain = {
				add_province_modifier = {
					name = shadow_plane_adaptability_flux_wastes
					duration = -1
				}
			}
		}
	}
