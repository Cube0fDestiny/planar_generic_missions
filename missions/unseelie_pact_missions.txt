unseelie_missions = {
	slot = 1
	generic = no
	has_country_shield = yes
	potential = {
		AND = {
			religion = unseelie_court
			capital_scope = { continent = feyrealm_continent }
		}
	}
	
	unseelie_show_of_strength_mission = {
		icon = mission_cannons_firing
		required_missions = {  }
		trigger = {
			war_score = 99
		}
		effect = {
			change_estate_land_share = {
				estate = all
				share = -5
			}
			add_country_modifier = {
				name = unseelie_display_of_power
				duration = -1
			}
		}
	}
	
	unseelie_ruthless_efficiency_mission = {
		icon = mission_impaled_soldies
		required_missions = { unseelie_show_of_strength_mission }
		trigger = {
			overextension_percentage = 1
		}
		effect = {
			add_country_modifier = {
				name = "unseelie_ruthless_efficiency"
				duration = -1
			}
			add_adm_power = 100
		}
	}
	
	unseelie_aggressive_diplomacy_mission = {
		icon = mission_oceania_assemble_an_army
		required_missions = { unseelie_ruthless_efficiency_mission }
		trigger = {
			any_country = { coalition_target = ROOT }
			custom_trigger_tooltip = {
				tooltip = unseelie_aggressive_diplomacy_tooltip
				any_rival_country = { humiliated_by = ROOT }
			}
		}
		effect = {
			add_country_modifier = {
				name = "unseelie_aggressive_diplomacy"
				duration = -1
			}
			every_country = {
				limit = {
					capital_scope = { continent = feyrealm_continent }
				}
				reverse_add_aggressive_expansion = {
					who = ROOT
					value = -25
					apply_calc = no
				}
			}
		}
	}
	
	unseelie_centralize_the_state_mission = {
		icon = mission_establish_the_prikazy
		required_missions = { unseelie_aggressive_diplomacy_mission }
		trigger = {
			NOT = {
				estate_influence = {
					estate = estate_nobles
					influence = 50
				}
			}
			NOT = {
			estate_influence = {
					estate = estate_church
					influence = 50
				}
			}
			NOT = {
				estate_influence = {
					estate = estate_burghers
					influence = 50
				}
			}
			crown_land_share = 50
		}
		effect = {
			remove_country_modifier = unseelie_display_of_power
			add_country_modifier = {
				name = "unseelie_centralized_state"
				duration = -1
			}
		}
	}
	
	unseelie_dominance_mission = {
		icon = mission_kowtow
		required_missions = { unseelie_centralize_the_state_mission }
		trigger = {
			num_of_owned_provinces_with = {
				value = 200
				continent = feyrealm_continent
			}
		}
		effect = {
			add_mil_power = 300
			add_adm_power = 100
			add_dip_power = 100
			remove_country_modifier = unseelie_aggressive_diplomacy
			remove_country_modifier = unseelie_ruthless_efficiency
			add_country_modifier = {
				name = "unseelie_ascendant"
				duration = -1
			}
			hidden_effect = {
				feyrealm_continent = {
					type = all
					limit = { religion = seelie_court }
					random = { 
						chance = 60 
						change_religion = unseelie_court
					}
				}
			}
		}
	}

	unseelie_absolute_rule_mission = {
		icon = mission_refine_the_carolean
		required_missions = { unseelie_dominance_mission }
		trigger = {
			absolutism = 80
		}
		effect = {
			remove_country_modifier = unseelie_centralized_state
			add_country_modifier = {
				name = "unseelie_absolute_rule"
				duration = -1
			}
			hidden_effect = {
			}
		}
	}
}
